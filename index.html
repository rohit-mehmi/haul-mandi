// Haul Mandi backend // - Redirects visitors from your site to your Meesho affiliate links // - Lets you add a new product by pasting its affiliate link // (auto-fetches the title/image/price for previews) const express = require("express"); const cors = require("cors"); const cheerio = require("cheerio"); const { createClient } = require("@supabase/supabase-js"); const app = express(); app.use(cors()); app.use(express.json()); // --- Supabase connection (uses environment variables set on Render) --- const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_SECRET_KEY ); // --- Health check (so you can confirm the server is alive) --- app.get("/", (req, res) => { res.send("Haul Mandi backend is running."); }); // --- 1. REDIRECT: haulmandi.com/go/meesho/ -> affiliate link --- app.get("/go/meesho/:productId", async (req, res) => { const { productId } = req.params; const { data, error } = await supabase .from("meesho_links") .select("affiliate_url") .eq("product_id", productId) .single(); if (error || !data) { return res.status(404).send("Link not found."); } return res.redirect(302, data.affiliate_url); }); // --- 2. ADD A NEW PRODUCT: paste an affiliate link, we fetch title/image/price --- // Protected: requires the correct admin key, so only you can add products. app.post("/api/links", async (req, res) => { try { const providedKey = req.headers["x-admin-key"]; if (!process.env.ADMIN_KEY || providedKey !== process.env.ADMIN_KEY) { return res.status(401).json({ error: "Unauthorized — incorrect or missing admin key." }); } const { product_id, affiliate_url, title, image_url, price } = req.body; if (!product_id || !affiliate_url) { return res.status(400).json({ error: "product_id and affiliate_url are required" }); } // If you typed in title/image/price yourself, use those directly. // Otherwise, try to auto-fetch them (Meesho sometimes blocks this). let finalTitle = title || null; let finalImage = image_url || null; let finalPrice = price || null; let fetchBlocked = false; if (!finalTitle || !finalImage) { const preview = await fetchProductPreview(affiliate_url); if (preview.blocked) { fetchBlocked = true; } else { finalTitle = finalTitle || preview.title; finalImage = finalImage || preview.image; finalPrice = finalPrice || preview.price; } } const { data, error } = await supabase .from("meesho_links") .upsert( { product_id, affiliate_url, title: finalTitle, image_url: finalImage, price: finalPrice, }, { onConflict: "product_id" } ) .select() .single(); if (error) { return res.status(500).json({ error: error.message }); } return res.json({ success: true, product: data, fetchBlocked }); } catch (err) { console.error(err); return res.status(500).json({ error: "Something went wrong fetching the product preview." }); } }); // --- 3. LIST ALL PRODUCTS (useful for showing them on your site) --- app.get("/api/links", async (req, res) => { const { data, error } = await supabase .from("meesho_links") .select("*") .order("created_at", { ascending: false }); if (error) { return res.status(500).json({ error: error.message }); } return res.json(data); }); // --- Helper: fetch title/image/price from a product page using Open Graph tags --- async function fetchProductPreview(url) { const response = await fetch(url, { redirect: "follow", headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0 Safari/537.36", }, }); const html = await response.text(); const $ = cheerio.load(html); const title = $('meta[property="og:title"]').attr("content") || $("title").text() || ""; const image = $('meta[property="og:image"]').attr("content") || null; // Price isn't always in meta tags — this looks for common price patterns as a fallback. const priceMatch = html.match(/₹\s?[\d,]+/); const price = priceMatch ? priceMatch[0] : null; // Detect common block/error pages instead of treating them as real product data const lowerTitle = title.toLowerCase(); const blocked = !response.ok || !title || lowerTitle.includes("access denied") || lowerTitle.includes("attention required") || lowerTitle.includes("just a moment") || lowerTitle.includes("captcha"); if (blocked) { return { title: null, image: null, price: null, blocked: true }; } return { title, image, price, blocked: false }; } const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
